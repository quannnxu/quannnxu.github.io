7.7 配置namenode进程以warehouse001启动

        <name>fs.defaultFS</name>
        <value>hdfs://warehouse001:9000</value>
    </property>
</configuration>

        <name>dfs.namenode.secondary.https-address</name>

1.sql行转列、列转行

我们首先先通过一个学生成绩表(下面简化了些)来形象了解下行转列

CREATE  TABLE mydatabasedb.StudentScores
(
UserName VARCHAR(20),     
Subject VARCHAR(30),
Score int
)

INSERT INTO mydatabasedb.StudentScores SELECT 'Nick', '语文', 80; 
INSERT INTO mydatabasedb.StudentScores SELECT 'Nick', '数学', 90;
INSERT INTO mydatabasedb.StudentScores SELECT 'Nick', '英语', 70;
INSERT INTO mydatabasedb.StudentScores SELECT 'Nick', '生物', 85;
INSERT INTO mydatabasedb.StudentScores SELECT 'Kent', '语文', 80;
INSERT INTO mydatabasedb.StudentScores SELECT 'Kent', '数学', 90;
INSERT INTO mydatabasedb.StudentScores SELECT 'Kent', '英语', 70; 
INSERT INTO mydatabasedb.StudentScores SELECT 'Kent', '生物', 85;

select  * from mydatabasedb.StudentScores

select 
UserName,
max(case Subject when '语文' then score else 0 end) as '语文',
max(case Subject when '数学' then score else 0 end) as '数学',
max(case Subject when '英语' then score else 0 end) as '英语',
max(case Subject when '生物' then score else 0 end) as '生物'
from mydatabasedb.StudentScores
group by UserName

列转行:

实现:

select * from (
    select name,'语文' as 科目,语文 as 分数 from mydatabasedb.student
    union all
    select name,'数学' as 科目,数学 as 分数 from mydatabasedb.student
    union all
    select name,'物理' as 科目,物理 as 分数 from mydatabasedb.student
) a

理解：把语文列装换为分数列，本身列头没有了，使用字符来填充列头，代表本身列具备的含义。

单个示例:

select  '语文' as 科目, 语文 as 分数 from mydatabasedb.student
