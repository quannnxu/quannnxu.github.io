#哪些部门的哪些职业的薪水和,最高1位的职业是什么?
#1.每个部门的每个职业的薪水和

create view sal
as 
select
deptno,job,
sum(sal+ifnull(comm,0)) as salary
from emp 
group by deptno,job

select * from sal;

select 
a.*
from sal a 
where 
(
select count(*) from sal b
where a.deptno=b.deptno 
and a.salary<b.salary        --自己关联自己,找到比自己大个数为0的,也就是最大的
)=0
order by a.deptno;
