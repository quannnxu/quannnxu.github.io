1.求每个域名截止到每个月的访问数,最大单月访问数,累计到该月的访问量.

create table access_tmp1 as
select 
domain,date_format(visit_time,'yyyy-MM') month,sum(num) pv
from user_click
group by domain,date_format(visit_time,'yyyy-MM');

1.1 第一种自连接:

create table access_tmp2 as
select 
a.domain a_domain,
a.month a_month,
a.pv a_pv,
b.domain b_domain,
b.month b_month,
b.pv b_pv
from access_tmp1 a join access_tmp1 b
on a.domain = b.domain;

访问到该月就是从头到当前月的,即a.month <= b.month

select 
t.b_domain,t.b_month,t.b_pv,
max(t.a_pv) max_pv,
sum(t.a_pv) sum_pv
from (
select * from access_tmp2 where a_month <= b_month
) t
group by t.b_domain,t.b_month,t.b_pv;

1.2 第二种窗口函数:

 select 
 domain,
 month,
 pv,
 max(pv) over(partition by domain order by month) max_pv,
 sum(pv) over(partition by domain order by month) sum_pv
 from access_tmp1;

可选参数:

(ROWS | RANGE) BETWEEN (UNBOUNDED | [num]) PRECEDING AND ([num] PRECEDING | CURRENT ROW | (UNBOUNDED | [num]) FOLLOWING)
窗口范围:UNBOUNDED 最前面    CURRENT ROW 到当前行    FOLLOWING 往后取
(ROWS | RANGE) BETWEEN CURRENT ROW AND (CURRENT ROW | (UNBOUNDED | [num]) FOLLOWING)
(ROWS | RANGE) BETWEEN [num] FOLLOWING AND (UNBOUNDED | [num]) FOLLOWING
sum(pv) over(partition by domain order by month rows BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) max_pv
从最前开始取到当前行
sum(pv) over(partition by domain order by month rows BETWEEN 3 PRECEDING AND CURRENT ROW ) max_pv
向前取一条
sum(pv) over(partition by domain order by month rows BETWEEN 3 PRECEDING AND 1 FOLLOWING ) max_pv
向前取三条向后取一条
sum(pv) over(partition by domain order by month rows BETWEEN CURRENT ROW AND  UNBOUNDED FOLLOWING) max_pv;
当前行取到最后一行

2.窗口函数:

2.1 NTILE():切片功能,把分区内的数据切分

select 
domain,
month,
pv,
ntile(2) over(partition by domain order by month) an1,
ntile(3) over(partition by domain order by month) an2
from access_tmp1
order by domain;

2.1 ROW_NUMBER():给窗口函数加上编号,一般用于排名,排名不会重复

select 
domain,
month,
pv,
ROW_NUMBER() over(partition by domain order by PV) an1
from access_tmp1;

2.3 RANK():相同排名编号一样,下一个排名会加上相同排名的个数

2.4 DENSE_RANK():相同排名编号一样,下一个排名按顺序加1

select 
domain,
month,
pv,
ROW_NUMBER() over(partition by domain order by PV) an1,
RANK() over(partition by domain order by PV) an2,
DENSE_RANK() over(partition by domain order by PV) an3
from access_tmp1;

2.5 CUME_DIST():小于等于当前行的行数 / 分组内的总行数 用来求占比

select 
domain,
month,
pv,
CUME_DIST() over(order by pv) rn1,
CUME_DIST() over(partition by domain order by pv) an2
from access_tmp1;

2.6 PERCENT_RANK(): 分组内当前行的rank-1 / 分组内的总行数 -1

select 
domain,
month,
pv,
PERCENT_RANK() over(order by pv) rn1,
PERCENT_RANK() over(partition by domain order by pv) an2
from access_tmp1;

2.7 lag(domain,1,0):取到上一条记录,取不到就默认为0

select 
domain,
month,
pv,
lag(domain,1,'-----') over(partition by domain order by pv) pre_domain,
lag(domain,2,'-----') over(partition by domain order by pv) pre_domain2
from 
access_tmp1;

2.8 lead(domain,1,0):取到先一条记录,取不到就默认为0

select 
domain,
month,
pv,
lead(domain,1,'-----') over(partition by domain order by pv) pre_domain,
lead(domain,2,'-----') over(partition by domain order by pv) pre_domain2
from 
access_tmp1;

2.9 first_value():取到当前分组的第一条

select 
domain,
month,
pv,
first_value(pv) over(partition by domain order by pv) pre_domain
from 
access_tmp1;

2.10 last_value():取到当前分组到现在为止的最后一条记录.

select 
domain,
month,
pv,
last_value(pv) over(partition by domain order by pv) pre_domain
from 
access_tmp1;

3.1 找参考系,先排序,时间减去排名为连续开始登陆时间

select 
name,
day,
rn,
date_sub(concat_ws('-',substr(day,0,4),substr(day,5,2),substr(day,7,2)),rn) date_diff

select 
name,
visit_time day,
row_number() over(partition by name order by visit_time) rn
from
visit) t;

3.2 分组求出连续登陆次数:

select
name,
count(1) times
from
(select 
name,
day,
rn,
date_sub(concat_ws('-',substr(day,0,4),substr(day,5,2),substr(day,7,2)),rn) date_diff
from(
select 
name,
visit_time day,
row_number() over(partition by name order by visit_time) rn
from
visit) a) b
group by name,date_diff;

result:

| A     | 3      |

3.3 去重求出最大登陆次数:

select 
name,
max(times)
from (select
name,
count(1) times
from
(select 
name,
day,
rn,
date_sub(concat_ws('-',substr(day,0,4),substr(day,5,2),substr(day,7,2)),rn) date_diff
from(
select 
name,
visit_time day,
row_number() over(partition by name order by visit_time) rn
from
visit) a) b
group by name,date_diff) c
group by name;

4.Hive参数

hive.fetch.task.conversion参数 控制是否执行MapReduce作业

  1. none: Disable hive.fetch.task.conversion (value added in Hive 0.14.0 with HIVE-8389)

    禁用,所有作业都跑MR

  2. minimal: SELECT *, FILTER on partition columns (WHERE and HAVING clauses), LIMIT only

    select分区列filter不执行

  3. more: SELECT, FILTER, LIMIT only (including TABLESAMPLE, virtual columns)

hive.mapred.mode参数
  • Default Value:
    • Hive 0.x: nonstrict
    • Hive 1.x: nonstrict
    • Hive 2.x: strict (HIVE-12413)

严格模式下order by不能执行,笛卡尔积操作也不能执行

查询分区表时不加分区,也不能执行

hive.exec.parallel参数 决定作业是否可以并行跑(没有依赖的作业)
  • Default Value: false
  • Added In: Hive 0.5.0

集群资源足够时可以开启.

hive.auto.convert.join参数 是否自动转换成为mapjoin
