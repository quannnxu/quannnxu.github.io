class NamePartitioner(names:Array[String]) extends Partitioner &#123;

  val namesMap = mutable.HashMap[String, Int]()

  for (i <- 0 until(names.length)) &#123;
    //  把name放到Map中去
    namesMap(names(i)) = i
  &#125;
  override def numPartitions: Int = names.size

  override def getPartition(key: Any): Int = &#123;
    // 这里的key是process中返回的类型,所以类型是(String, String)
    val name = key.asInstanceOf[(String, String)]._1
    namesMap(name)

  def main(args: Array[String]): Unit = &#123;
    val sc = ContextUtils.getSparkContext(this.getClass.getSimpleName)

    val lines = sc.textFile("data/group/test.txt")
//      .foreach(println)

    val process = lines.map(x => &#123;
      val splits = x.split(" ")
      val name = splits(0)
      val cnt = splits(1)
      ((name, cnt), 1)
    &#125;).reduceByKey(_ + _)


    val names = process.map(_._1._1).distinct().collect()
    val partitioner = new NamePartitioner2(names)

//    implicit val ord = Ordering[Int].on[((String, String), Int)](X => -X._2)
    implicit val ord = new Ordering[(String, String, Int)] &#123;
      override def compare(x: (String, String, Int), y: (String, String, Int)): Int = y._3- x._3
    &#125;

    /**
     * 使用repartitionAndSortWithinPartitions重新分区并在分区内部排序

      .setKeyOrdering(ord)
