package com.scalademo.bigdata.OO

object ConstrustorApp &#123;
  def main(args: Array[String]): Unit = &#123;
    val person = new Person("xiaoming",30)
    println(s"name is $&#123;person.name&#125;,age is $&#123;person.age&#125;")

    //这里调用的是附属构造器,附属构造器是对主构造器的补充
//    val person2 = new Person("李四",30,"家里蹲")

//    val student = new Student("zhangsan",18,"cs")

  &#125;

/**
 * 底层生产的就是这样的:
 * public class Person &#123;
 *    private final java.lang.String name;
 *    private int age;
 *
 *    public java.lang.String name();
 *    public int age();
 *
 *    public void age_$eq(int);
 *
 *    public Person(java.lang.String, int);
 * &#125;
 */

/**
 *这里的参数不加修饰的话属性是创建不成功的
 *val与car的选用:修饰时全部用val,val满足不了时再改为var
 * class 类名(....)    在Scala中叫做主构造器
 *附属构造器
 */
class Person(val name:String,val age:Int)&#123;  //默认也把括号内定义的内容转化为属性

  println("Person的构造器被调用")

  var school = "PKU"

  println("Person的构造器调用结束")

  /**
   *附属构造器
   *  1.def this定义的叫附属构造器
   *  2.一个附属构造器的第一行必须要调用主构造器或者其他附属构造器
   *  3.这里的属性不需要val,var修饰
   */
  def this(name:String, age:Int, school:String)&#123;
    this(name,age)
    this.school = "xxx"
  &#125;

  /**
   *继承:父和子之间
   * 先调用父类构造器,再调用子类构造器
   * 重写关键字: override
   */
  class Student(name:String, age:Int, val major:String) extends Person(name,age) &#123;

    println("Student的构造器被调用")

    //var school = "PKU"
    //这里调用父类属性需要重写
    override val school = "nudt"

    //这里默认是有toString方法,这里也需要重写
    override def toString() ="子类的toString方法"

    println("Student的构造器调用结束")

  &#125;

&#125;
